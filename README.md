                                                import PropTypes from 'prop-types';


                                                function TodoItem( {todo, index, onChange} ) {

                                                    }

                                                    }

                                                    return (
                                                        <div>

                                                                <span className={classes.join(' ')}>
                                                                    <input 
                                                                    type='checkbox' 
                                                                    checked={todo.completed}
